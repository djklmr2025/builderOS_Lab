    -H "Authorization: Bearer $SECRET_MASTER_TOKEN" \

Node (fetch):
  await fetch("http://HOST:PORT/aida/gateway", {
    method: "POST",
    headers: {
      "Authorization": "Bearer " + process.env.SECRET_MASTER_TOKEN,
      "Content-Type": "application/json"
    },
    body: JSON.stringify({"agent_id":"puter","action":"generate","kind":"report","params":{"topic":"ARKAIOS"}})
  }).then(r=>r.json());
