    <script>
        // Configuraci√≥n inicial
        const ADMIN_PIN = "1234"; // Cambiar esto en producci√≥n
        const CODIGOS_VALIDOS = {
            "ARK100ABCDEF1234": { valor: 100, valido: true },
            "ARK500GHIJKL5678": { valor: 500, valido: true },
            "ARK1000MNOPQR9012": { valor: 1000, valido: true }
        };
        
        // Elementos del DOM
        const authBox = document.getElementById('auth-box');
        const appContent = document.getElementById('app-content');
        const pinInput = document.getElementById('pin-input');
        const authButton = document.getElementById('auth-button');
        const statusMessage = document.getElementById('status-message');
        const logoutBtn = document.getElementById('logout-btn');
        const validarBtn = document.getElementById('validar-btn');
        const codigoInput = document.getElementById('codigo-input');
        const resultadoValidacion = document.getElementById('resultado-validacion');
        const cuentaDestino = document.getElementById('cuenta-destino');

        // Mostrar mensaje de estado
        function showStatus(message, type = 'success') {
            statusMessage.textContent = message;
            statusMessage.className = 'status-message ' + type;
            statusMessage.style.display = 'block';
            
            setTimeout(() => {
                statusMessage.style.display = 'none';
            }, 5000);
        }

        // Autenticaci√≥n
        authButton.addEventListener('click', () => {
            const pin = pinInput.value.trim();
            
            if (pin === ADMIN_PIN) {
                authBox.style.display = 'none';
                appContent.style.display = 'block';
                showStatus('Acceso concedido', 'success');
            } else {
                showStatus('PIN incorrecto', 'error');
                pinInput.value = '';
            }
        });

        // Validar c√≥digo
        validarBtn.addEventListener('click', () => {
            const codigo = codigoInput.value.trim().toUpperCase();
            const cuenta = cuentaDestino.value;
            
            if (!cuenta) {
                showStatus('Seleccione una cuenta', 'error');
                return;
            }
            
            if (codigo.length !== 16) {
                showStatus('El c√≥digo debe tener 16 d√≠gitos', 'error');
                return;
            }
            
            if (CODIGOS_VALIDOS[codigo] && CODIGOS_VALIDOS[codigo].valido) {
                resultadoValidacion.innerHTML = `
                    <h3>C√≥digo Validado</h3>
                    <div class="rate-display" style="color: var(--success);">‚úì V√ÅLIDO</div>
                    <p>Valor: ${CODIGOS_VALIDOS[codigo].valor} AEIO-MR</p>
                    <p>El cr√©dito ha sido asignado a ${cuenta}</p>
                `;
                resultadoValidacion.style.display = 'block';
                CODIGOS_VALIDOS[codigo].valido = false;
                showStatus('C√≥digo canjeado exitosamente', 'success');
            } else {
                resultadoValidacion.innerHTML = `
                    <h3>C√≥digo Inv√°lido</h3>
                    <div class="rate-display" style="color: var(--danger);">‚úó INV√ÅLIDO</div>
                    <p>El c√≥digo no existe o ya fue canjeado</p>
                `;
                resultadoValidacion.style.display = 'block';
                showStatus('C√≥digo no v√°lido', 'error');
            }
        });

        // Cerrar sesi√≥n
        logoutBtn.addEventListener('click', () => {
            location.reload();
        });
    </script>
