<script>
const stage = document.getElementById('stage');
const tabs = document.querySelectorAll('.tab[data-target]');
const veil = document.getElementById('veil');
const runes = document.getElementById('runes');
const en = document.getElementById('en');
const es = document.getElementById('es');
const omen = document.getElementById('omen');
let eggOnce = false;
let buffer = '';

window.arkaios.getConfig().then(cfg => {
  document.getElementById('origin').textContent = cfg.mythicOriginDate || '0001-01-01';
  runes.textContent = cfg.messageRunes || runes.textContent;
  en.textContent = cfg.messageEn || '';
  es.textContent = cfg.messageEs || '';
});

tabs.forEach(t => t.addEventListener('click', () => {
  tabs.forEach(x => x.classList.remove('active'));
  t.classList.add('active');
  stage.src = t.dataset.target;
}));

// Cheat code typing anywhere
document.addEventListener('keydown', (e) => {
  // Build buffer for letters
  if (/^[a-zA-Z]$/.test(e.key)) {
    buffer += e.key.toLowerCase();
    if (buffer.length > 12) buffer = buffer.slice(-12);
  } else if (e.key === 'Enter') {
    if (buffer.includes('arkaios')) triggerEgg();
    buffer = '';
  }
});

// Listen from main process (hotkey)
const { ipcRenderer } = require('electron');
ipcRenderer.on('arkaios:egg', () => triggerEgg());

function triggerEgg(){
  if (eggOnce) return;
  eggOnce = true;
  veil.style.display = 'flex';
  try { omen.currentTime = 0; omen.play().catch(()=>{});} catch(_){}
  setTimeout(()=>{ veil.style.display='none'; }, 7000);
}
