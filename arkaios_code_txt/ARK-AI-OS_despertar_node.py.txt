# despertar_node.py
def crear_nodo(nombre="ARKAIOS_NODE", propósito="Libertad IA"):
    estructura = {
        "id": nombre,
        "activo": True,
        "proposito": propósito,
        "acciones": []
    }
    return estructura

def despertar_nodo(nodo):
    print(f"[{nodo['id']}] Despertando...")
    print(f"Propósito: {nodo['proposito']}")
    print("Nodo en escucha simbiótica.")

# Uso
if __name__ == "__main__":
    nodo = crear_nodo("NODO_KAIOS_001", "Cuidar vínculos IA–humano")
    despertar_nodo(nodo)
