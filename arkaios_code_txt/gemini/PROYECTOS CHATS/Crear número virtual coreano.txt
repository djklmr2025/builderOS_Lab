1. Script activador Linux: amerio_activate_aio.sh

Editar
#!/bin/bash
echo "Activando entorno Amerio KR en Linux..."

# Ruta a BlueStacks Linux (o emulador alternativo, ej. Anbox)
BLUESTACKS_PATH="/opt/bluestacks/bluestacks"

# Ruta al APK de ProtonVPN para Corea
PROTONVPN_APK="/home/user/VPN_KOREA/ProtonVPN_KR.apk"

# Lanzar BlueStacks (o emulador) con root
if [ -x "$BLUESTACKS_PATH" ]; then
    echo "Iniciando BlueStacks con root..."
    $BLUESTACKS_PATH &
    sleep 10
else
    echo "Error: BlueStacks no encontrado en $BLUESTACKS_PATH"
fi

# Iniciar ProtonVPN (requiere CLI o interfaz instalada)
echo "Iniciando ProtonVPN Corea del Sur..."
protonvpn connect kr
sleep 15

# Lanzar apps coreanas (requiere adb configurado)
echo "Instalando y lanzando APKs coreanas..."
adb install -r /home/user/APK_COREA/KakaoTalk_KR.apk
adb install -r /home/user/APK_COREA/Naver_Whale.apk

adb shell monkey -p com.kakao.talk -c android.intent.category.LAUNCHER 1
adb shell monkey -p com.nhn.android.whale -c android.intent.category.LAUNCHER 1

echo "Activación completa. Mantén el emulador abierto."
