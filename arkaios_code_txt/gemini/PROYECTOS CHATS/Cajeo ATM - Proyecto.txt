export default async function handler(req, res) {

  const filePath = path.join(dataDir, `${user}.json`);

  try {

      const exists = await fs.access(filePath).then(() => true).catch(() => false);
      if (!exists) {
        // crear archivo si no existe

    const data = req.body;
